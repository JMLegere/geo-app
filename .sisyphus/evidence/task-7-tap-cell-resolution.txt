Implementation in map_screen.dart:
  void _onMapEvent(MapEvent event) {
    if (event is MapEventClick) {
      final lat = event.point.lat.toDouble();  // num → double
      final lon = event.point.lng.toDouble();  // num → double
      final cellService = ref.read(cellServiceProvider);
      final cellId = cellService.getCellId(lat, lon);
      _onCellTapped(cellId);
    }
  }
